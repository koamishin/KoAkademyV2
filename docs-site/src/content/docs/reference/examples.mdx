## Frontend Page Component

A typical Inertia page in `resources/js/pages/Dashboard.vue`:

```vue
<script setup lang="ts">
import AppLayout from '@/layouts/AppLayout.vue';
import { Head } from '@inertiajs/vue3';
import { dashboard as dashboardRoute } from '@/routes';

defineProps<{
    user: { id: number; name: string; email: string };
}>();

const breadcrumbs = [
    { title: 'Dashboard', href: dashboardRoute().url },
];
</script>

<template>
    <Head title="Dashboard" />

    <AppLayout :breadcrumbs="breadcrumbs">
        <div class="py-12">
            <div class="mx-auto max-w-7xl sm:px-6 lg:px-8">
                <div
                    class="overflow-hidden bg-white p-6 shadow-xl sm:rounded-lg dark:bg-zinc-900"
                >
                    Welcome back, {{ user.name }}!
                </div>
            </div>
        </div>
    </AppLayout>
</template>
```

## Backend Route Definition

Defining a route that is accessible via Wayfinder:

```php
// routes/web.php
use Inertia\Inertia;
use Illuminate\Support\Facades\Route;

Route::middleware(['auth', 'verified'])->group(function () {
    Route::get('/dashboard', function () {
        return Inertia::render('Dashboard');
    })->name('dashboard');
});
```

After saving, regenerate the Wayfinder types:

```bash
php artisan wayfinder:generate --with-form
```

Then use it in Vue:

```ts
import { dashboard } from '@/routes';
// dashboard()  →  { url: '/dashboard', method: 'get' }
```

## Wayfinder Form Submission

```ts
import { store as storePost } from '@/actions/App/Http/Controllers/PostController';
import { useForm } from '@inertiajs/vue3';

const form = useForm({
    title: '',
    body: '',
});

const submit = () => {
    storePost.form().submit({
        title: form.title,
        body: form.body,
    });
};
```

## Filament Resource Skeleton

```php
namespace App\Filament\Resources;

use App\Filament\Resources\PostResource\Pages;
use App\Models\Post;
use Filament\Forms;
use Filament\Forms\Form;
use Filament\Resources\Resource;
use Filament\Tables;
use Filament\Tables\Table;

class PostResource extends Resource
{
    protected static ?string $model = Post::class;
    protected static ?string $navigationIcon = 'heroicon-o-document-text';

    public static function form(Form $form): Form
    {
        return $form->schema([
            Forms\Components\TextInput::make('title')->required(),
            Forms\Components\RichEditor::make('body')->required(),
        ]);
    }

    public static function table(Table $table): Table
    {
        return $table->columns([
            Tables\Columns\TextColumn::make('title'),
            Tables\Columns\TextColumn::make('published_at')->dateTime(),
        ]);
    }

    public static function getPages(): array
    {
        return [
            'index'  => Pages\ListPosts::route('/'),
            'create' => Pages\CreatePost::route('/create'),
            'edit'   => Pages\EditPost::route('/{record}/edit'),
        ];
    }
}
```

## Pest Test Snippets

```php
use App\Models\User;

it('shows the dashboard to authenticated users', function () {
    $user = User::factory()->create();

    $this->actingAs($user)
        ->get('/dashboard')
        ->assertOk()
        ->assertInertia(fn ($page) => $page
            ->component('Dashboard')
            ->where('user.email', $user->email));
});

it('redirects guests to login', function () {
    $this->get('/dashboard')->assertRedirect('/login');
});
```

## Module Service Provider

```php
namespace Modules\Library\Providers;

use Illuminate\Support\ServiceProvider;

class LibraryServiceProvider extends ServiceProvider
{
    public function register(): void {}

    public function boot(): void
    {
        $this->loadRoutesFrom(__DIR__ . '/../../routes/web.php');
        $this->loadMigrationsFrom(__DIR__ . '/../../database/migrations');
        $this->loadViewsFrom(__DIR__ . '/../../resources/views', 'library');
    }
}
```
